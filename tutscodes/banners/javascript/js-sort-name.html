Sort an Array of Objects

By Name Value - ASC Order 


users = users.sort((a, b) => {
  return (a.name.toLowerCase() > b.name.toLowerCase()) ? 1 : -1;
});

By Name Value - DESC Order 


users = users.sort((a, b) => {
  return (a.name.toLowerCase() < b.name.toLowerCase()) ? 1 : -1;
});
